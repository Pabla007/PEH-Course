def decrypt(puzzle,n):
puzzle = puzzle.lower()
letters="abcdefghijklmnopqrstuvwxyz"
decrypted_message = ""

for ch in puzzle:

if ch in letters:
position = letters.find(ch)
new_pos = (position - n) % 26
new_char = letters[new_pos]
decrypted_message += new_char
else:
decrypted_message += ch

return decrypted_message
