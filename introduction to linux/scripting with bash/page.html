#! /bin/bash

#cat ip.txt | grep "64 bytes" | cut -d " " -f 4 | tr -d ":"
if [ "$1" == "" ]
then
echo "You forgot an IP address!"
echo "Syntax: ./ipweep.sh 192.168.136"

else
for ip in `seq 1 254`; do
ping -c 1 $1.$ip | grep "64 bytes" | cut -d " " -f 4 | tr -d ":" &b
done
fi

#Argument One (i.e $1)
#./ipsweep.sh 192.168.136

& will be used to run multiple instance of the command in the for loop.
;will be usedd to run only one instance of the command in the for loop.
